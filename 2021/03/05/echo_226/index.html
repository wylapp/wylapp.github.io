快读-模板

2021年3月5日

其实就是省去了缓冲区同步的时间，把一切输入当作字符处理

#include <cmath>
#include <algorithm>
#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cctype>
using namespace std;
inline int read()
{
    int x=0,f=1;char ch=getchar();
    while (!isdigit(ch)){if (ch=='-') f=-1;ch=getchar();}
    while (isdigit(ch)){x=x*10+ch-48;ch=getchar();}
    return x*f;
}
int main()
{
    int sx = read();
    cout<<sx;
    return 0;
}
